import 'package:firebase_auth/firebase_auth.dart';
import 'package:flutter/material.dart';

class AuthProvider with ChangeNotifier {
  User? _user;
  String? _role; // 'student', 'parent', 'teacher', 'admin'

  User? get user => _user;
  String? get role => _role;

  AuthProvider() {
    FirebaseAuth.instance.authStateChanges().listen((User? user) {
      _user = user;
      // Later: fetch role from Firestore after login
      if (user != null) _role = 'student'; // demo  replace with real fetch
      notifyListeners();
    });
  }

  Future<void> signIn(String email, String password) async {
    try {
      await FirebaseAuth.instance.signInWithEmailAndPassword(
        email: email,
        password: password,
      );
    } catch (e) {
      rethrow;
    }
  }

  Future<void> signOut() async {
    await FirebaseAuth.instance.signOut();
  }

  // Add signUp, role assignment later
}

// Show/hide sections
function showSection(id) {
  document.querySelectorAll('section').forEach(sec => sec.classList.remove('active'));
  document.getElementById(id).classList.add('active');

// Load data from localStorage
function loadData(key, defaultValue = []) {
  const data = localStorage.getItem(key);
  return data ? JSON.parse(data) : defaultValue;
}

function saveData(key, value) {
  localStorage.setItem(key, JSON.stringify(value));
}

// === Timetable ===
let timetable = loadData('timetable', [
  { time: "8:00 - 9:00", mon: "Math", tue: "English", wed: "Science", thu: "Social", fri: "Civic" },
  { time: "9:00 - 10:00", mon: "English", tue: "Math", wed: "ICT", thu: "Yoruba", fri: "PE" }
]);

function renderTimetable() {
  const tbody = document.querySelector('#timetableTable tbody');
  tbody.innerHTML = '';
  timetable.forEach(row => {
    const tr = document.createElement('tr');
    tr.innerHTML = `
      <td contenteditable="true" onblur="updateTimetable(this, '${row.time}', 'time')">${row.time}</td>
      <td contenteditable="true" onblur="updateTimetable(this, '${row.time}', 'mon')">${row.mon}</td>
      <td contenteditable="true" onblur="updateTimetable(this, '${row.time}', 'tue')">${row.tue}</td>
      <td contenteditable="true" onblur="updateTimetable(this, '${row.time}', 'wed')">${row.wed}</td>
      <td contenteditable="true" onblur="updateTimetable(this, '${row.time}', 'thu')">${row.thu}</td>
      <td contenteditable="true" onblur="updateTimetable(this, '${row.time}', 'fri')">${row.fri}</td>
    `;
    tbody.appendChild(tr);
  });
}

function updateTimetable(el, time, day) {
  const row = timetable.find(r => r.time === time);
  if (row) row[day] = el.textContent.trim();
  saveData('timetable', timetable);
}

Guide to hosting on GitHub Pages

Here's a clear, up-to-date step-by-step guide (as of 2026) to host your simple static website (the HTML + CSS + JS school portal we built earlier) on GitHub Pages for free. GitHub Pages is perfect for this it's reliable, fast, supports HTTPS automatically, and your site will be live at a URL like https://fadodun.github.io/fadodun-school-site/ (or a nicer custom domain later if you want).

Prerequisites

  • A GitHub account (free). If you don't have one go to https://github.com and sign up.
  • Your website files in a folder on your computer (index.html, style.css, script.js). If you haven't already, create a folder called fadodun-school-site and put the three files inside.

You have two main ways: Option A Easiest for beginners (no command line / Git needed) Use GitHub's web uploader Option B Better for future updates (using Git locally) Recommended if you plan to edit often

Option A: Upload Directly via Browser (No Software Install Needed)

  1. Log in to GitHub https://github.com
  2. Create a new repository
    • Click the + icon (top right) New repository
    • Repository name: fadodun-school-site (or any name you like, e.g. school-portal)
    • Description: optional (e.g. "Fadodun School Simple Portal")
    • Visibility: Public (required for free GitHub Pages)
    • Do not check "Add a README file" yet (we'll add files manually)
    • Click Create repository
  3. Upload your files
    • On the new repo page, click Add file Upload files
    • Drag & drop your three files: index.html, style.css, script.js (Or click "choose your files" and select them)
    • In the commit message box, write something like "Initial upload of school portal"
    • Click Commit changes
  4. Enable GitHub Pages
    • In your repo, click Settings (top tab, gear icon)
    • Scroll down (left sidebar) click Pages
    • Under Build and deployment Source: Select Deploy from a branch
    • Branch: Choose main (or whatever your default branch is)
    • Folder: Select /(root)
    • Click Save
  5. Wait 15 minutes GitHub will build & deploy. Refresh the Pages section you'll see a green box with your live URL, something like: https://fadodun.github.io/fadodun-school-site/ (If your username is fadodun and repo is fadodun-school-site)
  6. Visit your site Open that URL in a browser. Your school portal should appear!
    • If you see a 404 error wait a bit longer or check that index.html is in the root (not in a subfolder).
    • Clear browser cache (Ctrl+Shift+R) if needed.

Option B: Using Git Locally (Better for Editing & Updates Later)

If you want to edit files on your computer and push changes easily:

  1. Install Git (if not already): https://git-scm.com/downloads (Very quick install for Windows/Mac/Linux)
  2. Create the repo on GitHub as in Option A step 2 (but you can add README if you want).
  3. On your computer (in terminal / command prompt / PowerShell):
    text
    cd path/to/your/fadodun-school-site-folder
git init
git add .
git commit -m "Initial school portal files"
git branch -M main
git remote add origin https://github.com/fadodun/fadodun-school-site.git
git push -u origin main
    (Replace with your actual repo URL GitHub shows it after creating the repo)
  4. Enable Pages exactly as in Option A step 4.
  5. For future changes: Edit files
    text
    git add .
git commit -m "Updated timetable data"
git push
    Site updates automatically in ~1 minute.

Important Tips & Troubleshooting

  • Must have index.html in the root folder GitHub Pages looks for this file first.
  • URL format:
    • Project site: https://yourusername.github.io/repo-name/
    • If you name the repo exactly yourusername.github.io (e.g. fadodun.github.io), it becomes your personal site at https://fadodun.github.io (no repo name in URL)
  • Updates not showing? Wait 110 min, or force-refresh page (Ctrl+F5).
  • HTTPS is automatic & free.
  • Custom domain (e.g. school.fadodun.com): Possible later add in Pages settings + DNS records (I can guide if needed).
  • Data note: Your current site uses localStorage each visitor's browser saves their own timetable/news/etc. If you want shared data (everyone sees same announcements), we'll need to add Firebase later (still free tier).
  • Mobile friendly: Your site already is (thanks to the CSS media query) test on phone.
